           - **CRITICAL:** Project out `password_hash` field: `SetProjection(bson.D{{Key: "password_hash", Value: 0}})` — even though `json:"-"` prevents JSON serialization of PasswordHash, always project it out from MongoDB queries as belt-and-suspenders security
         - Find all users in CollUsers (no filter — admin sees all users regardless of role)

         - If result is nil (no users), return empty JSON array `[]` (not null)
         - Return 200 with JSON array
       ```go
       func ListUsers(database *mongo.Database) http.HandlerFunc {
           return func(w http.ResponseWriter, r *http.Request) {
               opts := options.Find().
                   SetSort(bson.D{{Key: "created_at", Value: -1}}).
                   SetProjection(bson.D{{Key: "password_hash", Value: 0}})

               cursor, err := database.Collection(db.CollUsers).Find(r.Context(), bson.D{}, opts)
               if err != nil {
                   http.Error(w, "failed to list users", http.StatusInternalServerError)
                   return
               }

               var users []models.User
               if err := cursor.All(r.Context(), &users); err != nil {
                   http.Error(w, "failed to decode users", http.StatusInternalServerError)
                   return
               }
               if users == nil {
                   users = []models.User{}
               }

               w.Header().Set("Content-Type", "application/json")
               json.NewEncoder(w).Encode(users)
           }
       }
       ```
