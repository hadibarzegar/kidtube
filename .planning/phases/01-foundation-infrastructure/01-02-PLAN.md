RUN go mod download

# Site API target
FROM builder AS site-api-build
RUN CGO_ENABLED=0 go build -o /bin/site-api ./cmd/site-api

COPY --from=admin-api-build /bin/admin-api /bin/admin-api

package db

// Connect establishes a MongoDB connection and returns the database handle.
// Uses mongo-driver v2 — the Connect function no longer requires a separate Ping.
func Connect(ctx context.Context, uri string) (*mongo.Database, error) {
    clientOpts := options.Client().ApplyURI(uri)
    client, err := mongo.Connect(clientOpts)
    if err != nil {
        return nil, fmt.Errorf("mongo connect: %w", err)
    }
    if err := client.Ping(ctx, nil); err != nil {
        return nil, fmt.Errorf("mongo ping: %w", err)
    }
    dbName := "kidtube"  // extract from URI or hardcode
    return client.Database(dbName), nil
}

// Collection name constants
